    <p>
        <span class="tag tag-blue">Lorem</span>
        <span class="tag tag-danger">Lorem</span>
        <span class="tag tag-warning">Lorem</span>
        <span class="tag tag-ok">Lorem</span>
        <span class="tag tag-info">Lorem</span>
        <span class="tag tag-dark">Lorem</span>
        <span class="tag tag-gray">Lorem</span>
        <span class="tag tag-light">Lorem</span>
    </p>
</div>
